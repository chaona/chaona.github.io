先下载支付宝电脑网站支付demo，目录结构如下：

把demo文件夹放到tp5的 vendor 文件夹下面，重命名为alipay：

配置 config.php 的支付参数，去蚂蚁金服商家中心找，或者使用沙箱测试环境。注意签名方式要和私钥、公钥的签名方式相同。

  1. <?php
  2. $config = array (  
  3.  //APPID。
  4.  'app_id' => "xxx",
  5.  
  6.   //商户私钥
  7.  'merchant_private_key' => "xxx",
  8.   
  9.  //异步通知地址
  10.   'notify_url' => "https://www.oyhdo.com/alipay/alinotify",
  11.   
  12.   //同步跳转
  13.   'return_url' => "https://www.oyhdo.com/alipay/alinotify",
  14.  
  15.  //编码格式
  16.   'charset' => "UTF-8",
  17.  
  18.  //签名方式
  19.   'sign_type'=>'RSA2',
  20.  
  21.   //支付宝网关 （我使用的沙箱环境）
  22.   'gatewayUrl' => "https://openapi.alipaydev.com/gateway.do",
  23.         //'gatewayUrl' => "https://openapi.alipay.com/gateway.do",
  24.  
  25.   //支付宝公钥,查看地址：https://openhome.alipay.com/platform/keyManage.htm 对应APPID下的支付宝公钥。
  26.  'alipay_public_key' => "xxx",
  27. );

设置 log.txt 权限为777，以便记录日志。

Alipay.php控制器：

  1. <?php
  2. namespace app\index\controller;
  3. use think\Controller;
  4. class Alipay extends Controller
  5. {
  6.     public function index(){
  7.         header("Content-type:text/html;charset=utf-8");
  8.         
  9.         $total_amount = input('post.total_amount');
  10.         if($total_amount){
  11.             //引入支付宝支付
  12.             require_once VENDOR_PATH.'/alipay/config.php';
  13.             require_once VENDOR_PATH.'/alipay/pagepay/service/AlipayTradeService.php';
  14.             require_once VENDOR_PATH.'/alipay/pagepay/buildermodel/AlipayTradePagePayContentBuilder.php';
  15.  
  16.             //商户订单号，商户网站订单系统中唯一订单号，必填
  17.             $out_trade_no = input('post.out_trade_no');
  18.  
  19.             //订单名称，必填
  20.             $subject = input('post.goods_name');
  21.  
  22.             //付款金额，必填
  23.             $total_amount = $total_amount;
  24.  
  25.             //商品描述，可空
  26.             $body = input('post.goods_body');
  27.  
  28.             //构造参数
  29.             $payRequestBuilder = new \AlipayTradePagePayContentBuilder();
  30.             $payRequestBuilder->setBody($body);
  31.             $payRequestBuilder->setSubject($subject);
  32.             $payRequestBuilder->setTotalAmount($total_amount);
  33.             $payRequestBuilder->setOutTradeNo($out_trade_no);
  34.  
  35.             //电脑网站支付请求
  36.             $aop = new \AlipayTradeService($config);
  37.             $response = $aop->pagePay($payRequestBuilder,$config['return_url'],$config['notify_url']);
  38.  
  39.             //输出表单
  40.             var_dump($response);
  41.         }else{
  42.             $out_trade_no = 'ALPAY'.date('YmdHis'); //订单号
  43.             $goods_name = '在线支付'; //商品名称
  44.             $goods_body = 'test'; //商品描述
  45.  
  46.             $this->assign('out_trade_no',$out_trade_no);
  47.             $this->assign('goods_name',$goods_name);
  48.             $this->assign('goods_body',$goods_body);
  49.             return view();
  50.         }
  51.     }
  52. }
